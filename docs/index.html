l <- function(lambda) {
  sum(dpois(cyclones$number, 
            lambda = lambda, 
            log = TRUE))
}

g <- Vectorize(l)

curve(g, 0, 15, axes = FALSE, col = "blue")
axis(1)
axis(2, las = 1, pos = 0)
abline(h = sum(dpois(cyclones$number, 
                     lambda = mean(cyclones$number), 
                     log = TRUE)
), 
lty = 2, 
col = "red")
abline(v = mean(cyclones$number),
       lty = 2,
       col = "red")
